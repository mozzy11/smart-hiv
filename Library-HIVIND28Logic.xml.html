// Included Libraries
using FHIR version '4.0.1'

include HIVCommon version '0.0.1' called HIC
include FHIRHelpers version '4.0.1'
include FHIRCommon called FC
include WHOCommon called WCom

// Indicator Definition
parameter &quot;Measurement Period&quot; Interval&lt;Date&gt; default Interval[@2023-01-01, @2023-01-30]

context Patient
/*
 * As defined by Member State
 */
define &quot;Initial Population&quot;:
  true


define &quot;numerator&quot;:
   (
    HIC.&quot;Patient Deceased before end of Measurement Period&quot;
   and HIC.&quot;Transfer Out at Facility during the measurement period&quot;
   and HIC.&quot;Lost to Follow Up during the measurement period&quot;
   )
   and ((
    //saying needs to not be counted this measurement period but should have counted this period or last period
    (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset before end of &quot;Measurement Period&quot;)
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot;)
	))
   and exists(HIC.&quot;HIV Treatment Active&quot; H
   where H.effective starts before end of &quot;Measurement Period&quot;
    and H.effective ends after start of &quot;Measurement Period&quot;
   )
   ) or
   (
    //Accounts for end of last period
    (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset before (end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;))
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
	))
   and exists(HIC.&quot;HIV Treatment Active&quot; H
   where H.effective starts before (end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
    and H.effective ends after (start of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
   )
   )
   )



 define &quot;denominator&quot;:
   ((
    //saying needs to not be counted this measurement period but should have counted this period or last period
    (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset before end of &quot;Measurement Period&quot;)
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot;)
	))
   and exists(HIC.&quot;HIV Treatment Active&quot; H
   where H.effective starts before end of &quot;Measurement Period&quot;
    and H.effective ends after start of &quot;Measurement Period&quot;
   )
   ) or
   (
    //Accounts for end of last period
    (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset before (end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;))
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
	))
   and exists(HIC.&quot;HIV Treatment Active&quot; H
   where H.effective starts before (end of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
    and H.effective ends after (start of &quot;Measurement Period&quot; - duration in days of &quot;Measurement Period&quot;)
   )
   )
   )
