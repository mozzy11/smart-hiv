 * As defined by Member State

  true


//Does this also filter out self HIV tests

define &quot;numerator&quot;:
    exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset during &quot;Measurement Period&quot;)
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot;
	and O.issued after start of &quot;Measurement Period&quot;)
	) 

define &quot;denominator&quot;:
    exists(HIC.&quot;HIV test resulted&quot; T
	where T.issued before end of &quot;Measurement Period&quot;
	and T.issued after start of &quot;Measurement Period&quot;)
