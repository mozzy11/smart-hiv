Countries with aggregate reporting need data collection forms that categorize those who initiate ART by the timing of their HIV diagnosis. This can result in some mismatch between numerator and denominator, as some who are diagnosed with HIV toward the end of the reporting period (and so counted in the denominator) may initiate ART after the reporting period (and so not counted in the numerator). // Indicator Definition
parameter &quot;Measurement Period&quot; Interval&lt;Date&gt; default Interval[@2023-01-01, @2023-01-30]

context Patient
/*
 * As defined by Member State
 */
define &quot;Initial Population&quot;:
  true


define &quot;numerator&quot;:
   (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset during &quot;Measurement Period&quot;)
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
    //during doesn't work
	where (O.issued before end of &quot;Measurement Period&quot; and
    O.issued after start of &quot;Measurement Period&quot;)
	))
   and exists(HE.&quot;On ART H.DE47&quot; H
   where H.effective starts before end of &quot;Measurement Period&quot;
    and (H.effective ends after start of &quot;Measurement Period&quot;
    or end of H.effective  ~  null)
   )

define &quot;denominator&quot;:
   (exists(HIC.&quot;HIV Positive Condition&quot; C
	where C.onset before end of &quot;Measurement Period&quot;)
	or 
	exists(HIC.&quot;HIV Positive Observation&quot; O
	where (O.issued before end of &quot;Measurement Period&quot; and
    O.issued after start of &quot;Measurement Period&quot;)
	))

/*
